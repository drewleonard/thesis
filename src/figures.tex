\begin{algorithm}
	\caption{Discover Interest Groups}\label{interests}
	\begin{algorithmic}[1]
		\Require $InterestGroups$, mapping group labels to keywords
		\State Initialize $Found \gets \{\}$
		\Procedure{FindInterests}{}
		\ForEach {$Interest \in Documents $}
		\ForEach{$InterestGroup \in InterestGroups$}
		\If{$Interest \cap InterestGroup$}
		\State $Found[Interest] \gets \{InterestGroup: +1\}$
		\EndIf{}
		\EndFor
		\EndFor
		\EndProcedure
		\Procedure{RankInterests}{}
		\ForEach {$Interest \in Found $}
		\State Initialize $MaxGroup \gets \{\}$
		\ForEach {$Title, Rank \in Interest $}
		\If{$Rank > MaxGroup.Rank$}
		\State $MaxGroup \gets {Title, Rank}$
		\EndIf
		\EndFor
		\State $Found[Interest] \gets MaxGroup.Title$
		\EndFor
		\EndProcedure
	\end{algorithmic}
\end{algorithm}
